  def test_age
    load_default_strings
    [
      [1, '1 yr', '1歳', '1 год'],
      [22, '22 yrs', '22歳', '22 года'],
      [27, '27 yrs', '27歳', '27 лет'],
    ].each do |a, en, ja, ru|
      set_language :en
      assert_equal en, l_age(a)
      set_language :ja
      assert_equal ja, l_age(a)
      set_language :ru
      assert_equal ru, l_age(a)
    end
  end
