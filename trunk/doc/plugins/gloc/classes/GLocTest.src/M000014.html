  def test_all_languages_have_values_for_helpers
    load_default_strings
    t= Time.local(2000, 9, 15, 11, 23, 57)
    GLoc.valid_languages.each {|l|
      set_language l
      0.upto(120) {|n| l_age(n)}
      l_date(t)
      l_datetime(t)
      l_time(t)
      [true,false].each{|v| l_YesNo(v); l_yesno(v) }
    }
  end
