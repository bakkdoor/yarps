    def set_kcode(charset=nil)
      _charset_required
      charset ||= get_charset(current_language)
      $KCODE= case charset
        when UTF_8 then 'u'
        when SHIFT_JIS then 's'
        when EUC_JP then 'e'
        else 'n'
        end
      _verbose_msg {"$KCODE set to #{$KCODE}"}
    end
